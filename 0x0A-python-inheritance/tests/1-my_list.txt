# 1-my_list.py

This module contains a class MyList that inherits from list.

Importing:
>>> MyList = __import__('1-my_list').MyList

Test:
>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)
>>> print(my_list)
[1, 4, 2, 3, 5]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[1, 4, 2, 3, 5]

>>> my_list = MyList(["Hello", "World", "School"])
>>> my_list.print_sorted()
['Hello', 'School', 'World']
